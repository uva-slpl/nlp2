	$$P(E|F) = \frac{P(E)P(F|E)}{P(F)}$$
	
	Inference
	
	$$\hat{E} = \argmax_E P(E)P(F|E)$$
	
	Estimation
	
	\begin{itemize}
		\item $P(E)$ $n$-gram LM
		\item $P(F|E)$ TM
